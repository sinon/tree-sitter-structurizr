  model {
    archetypes {
      hardwareSystem = element {
        metadata "Hardware System"
        tag "Hardware System"
      }
    }

    a = softwareSystem "A"
    b = hardwareSystem "B"

    a -> b "Gets data from"
  }
}

Archetype extensions
====================

workspace {
  model {
    archetypes {
      softwareSystem {
        tag "Default Tag"
      }
      sync = -> {
        tag "Synchronous"
      }
      https = --sync-> {
        technology "HTTPS"
        tag "HTTPS"
      }
    }

    a = softwareSystem "A"
    b = softwareSystem "B"
    a --https-> b "Makes API calls to"
  }
}
