=========================
Element selectors and groups
=========================

workspace {
  !identifiers hierarchical

  model {
    properties {
      "structurizr.groupSeparator" "/"
    }

    team = group "Team" {
      system = softwareSystem "System" {
        api = container "API" {
          worker = component "Worker"
        }
      }
    }

    !element system.api.worker {
      properties {
        "owner" "Team"
      }
    }

    !elements team {
      -> this "Uses"
    }
  }
}
